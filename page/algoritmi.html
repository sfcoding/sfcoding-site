                      void insertion_sort (int *A, int n) {
                        int i, j, z;
                        for (i = 1; i <; n; i++){
                          z = A[i], j = i-1;
                          while ( j >= 0 && A[j] > z ){
                            A[j+1] = A[j];
                            j--;
                          }
                          A[j+1] = z;
                        }
                        return;
                      }

                      void merge (int *A, int p, int q, int r){
                        int n1 = q - p + 1;
                        int n2 = r - q; //r - p + 1 - n1
                        int i, L[n1+1], R[n2+1];
                       
                        for (i=0; i<n1; i++)
                          L[i] = A[p+i-1];
                        L[n1] = INT_MAX;
                        for (i=0; i<n2; i++)
                          R[i] = A[q+i];
                        R[n2] = INT_MAX;
                       
                        i=0; int j=0, k;
                        for(k=p; k<r; k++){
                          if (L[i] < R[j]){
                            A[k] = L[i];
                            i++;
                          }
                          else {
                            A[k] = R[j];
                            j++;
                          }
                        }
                      }
                       
                      void merge_sort(int *A, int p, int r){
                          if (p<r){
                            int q = (r+p)/2;
                            merge_sort(A, p, q);
                            merge_sort(A, q+1, r);
                            merge(A, p, q, r);
                          }
                      }
                       
                      void quick_sort(int *A, int p, int r){
                        int pivot, j, i;
                        if(p<r){
                          pivot = p;
                          i = p;
                          j = r;
                          while(i<j){
                            while(A[i] <= A[pivot] && i < r)
                              i++;
                            while(A[j] > A[pivot])
                              j--;
                            if(i<<j)
                              swap(A+i, A+j);
                               }
                          swap(A+pivot, A+j);
                          quick_sort(A, p, j-1);
                          quick_sort(A, j+1, r);
                          }
                      }

                      void maxheapify (int*A, int i, int n) {
                        int t = i;
                        if(2*i <=n && A[i] < A[2*i])
                          t = 2*i;
                        if(2*i+1 <=n && A[2*i + 1] > A[t])
                          t = 2*i + 1;
                        if (i != t){
                          swap(A+t, A+i);
                          maxheapify(A, t, n);
                        }
                      }
                       
                      void buildmaxheap (int*A, int n) {
                        int i;
                        for(i= n/2; i > -1; i--)
                          maxheapify(A, i, n);
                      }
                       
                      void heap_sort(int *A, int n){
                        buildmaxheap(A,n);
                        int i;
                        for(i=n; i > 0; i--){
                          swap(A, A+i);
                          maxheapify(A, 0, i-1);
                        }
                      }

                      void swap(int*a, int*b){
                        int t = *a;
                        *a = *b;
                        *b = t;
                      }

                      void merge(vector  & v, uint p, uint q, uint r){
                       
                        uint s1 = q - p + 2;
                        uint s2 = r - q + 1;
                        uint i = 0, j = 0;
                       
                        vector  v1(s1);
                        vector  v2(s2);
                       
                        for(; i < s1 - 1 ; ++i)
                          v1.at(i) = v.at(p + i);
                       
                        v1.at(i) = numeric_limits::max();
                       
                        for(i = 0; i < s2 - 1; ++i)
                          v2.at(i) = v.at(q + i + 1);
                       
                        v2.at(i) = numeric_limits::max();
                       
                        i = 0;
                       
                        for(uint k = p; k <= r; ++k)
                          v.at(k) = v1.at(i) < v2.at(j) ? v1.at(i++) : v2.at(j++);
                       
                      }
                       
                      template 
                      void mergeSort(vector  & v, uint p, uint r){
                       
                        if(p >= r) return;
                        int q = floor((p + r) / 2);
                        mergeSort(v, p, q);
                        mergeSort(v, q+1, r);
                        merge(v, p, q, r);
                       
                      }
                       
                      template 
                      void mergeSort(vector  & v){
                        mergeSort(v, 0, v.size()-1);

                      template 
                      int partition(vector  &v, uint p, uint r){
                        int i = p;
                        for(uint j = p; j < r; ++j)
                            if(v.at(j) <= v.at( r ))
                              swap(v.at(j), v.at(i++));
                        swap(v.at(i), v.at( r ) );
                        return i;
                      }
                       
                       
                      template 
                      void quickSort(vector  &v, uint p, uint r){
                        if(p >= r) return;
                        int q = partition(v,p,r);
                        quickSort(v, p, q-1);
                        quickSort(v, q+1, r);
                      }

                        quickSort(v, 0, v.size()-1);

                          if(I[i] > limit)
                              limit = I[i];

                        vector  * O = new vector(size);
                       
                        for(i = 0; i <= limit; ++i)
                          v[i] = 0;
                       
                        for(i = 0; i < size; ++i)
                          v[I[i]]++;
                       
                        for(i = 1; i < limit; ++i)
                          v[i] += v[i-1];

                        {
                          O->at(v[I[i]] - 1) = I[i];
                          v[I[i]]--;
                        }
                       
                        I = *O;
                       
                        delete[] v;

                      void selectionSort(vector  & v){
                        for(int i = 0; i < v.size() - 1; i++)
                        {
                          int min = i;
                          for(int j = i+1; j < v.size(); j++)
                            if(v.at(j) < v.at(min))
                              min = j;
                          swap(v[min], v[i]);
                        }

                      class Heap{
                       
                        private:
                          vector  * v;
                          uint heapSize;
                       
                              public:
                       
                          static int parent(uint i){
                            return i/2;
                          }
                       
                          static int left(uint i){
                            return i*2 + 1;
                          }
                       
                          static int right(uint i){
                            return 2*(i+1);
                          }
                       
                          void maxHeapify(uint i){
                            uint max = i;
                            uint l = left(i);
                            uint r = right(i);
                            if(l < heapSize && v->at(max) < v->at(l))
                                max = l;
                            if(r < heapSize &&v->at(max) < v->at( r ))
                                max = r;
                            if(max != i){
                              swap(v->at(i), v->at(max));
                              maxHeapify(max);
                            }
                          } 
                       
                          void buildMaxHeap(){
                            for(uint i = floor(v->size()/2)+1; i-- > 0; )
                                 maxHeapify(i);
                          }
                       
                          static void sort(vector  &v){
                       
                            Heap * hp = new Heap(v);
                       
                            for(uint i = v.size() - 1; i > 0; --i)
                            {
                              swap(v.at(0), v.at(i));
                              --hp -> heapSize;
                              hp -> maxHeapify(0);
                            }
                       
                            delete hp;
                          }
                       
                          T maximum(){
                            return v->at(0);
                          }
                       
                          T extractMaximum(){
                            int max = v->at(0);
                            v->at(0) = v->at(--heapSize);
                            maxHeapify(0);
                            return max;
                          }
                       
                          void increaseToKey(uint i, uint k)
                          {
                            if(k < v->at(i))
                            {
                              cout << "k dev'essere maggiore del vecchio valore!";
                              return;
                            }
                            v->at(i) = k;
                            uint par = parent(i)+1;
                            while(par-- > 0 && v->at(par) <v->at(i))
                            {
                              swap(v->at(i), v->at(par));
                              i = par;
                              par = parent(par)+1;
                            }
                          }
                       
                          void insert(T k){
                            v->push_back(numeric_limits::min());
                            increaseToKey(heapSize++, k);
                          }
                       
                          Heap(vector  &v){
                            this->v = &v;
                            this->heapSize = v.size();
                            buildMaxHeap();
                          }
                       
                      };

                      package mergesort;
                       
                      public class MergeSort {
                       
                          static public void Sort(int[] A) {
                              SortCore(A,0,A.length-1);
                          }
                       
                          static private void SortCore(int[] A,int p,int r) {
                              if(p<r) {
                                  int q=(p+r)/2;
                                  SortCore(A,p,q);
                                  SortCore(A,q+1,r);
                                  Merge(A,p,q,r);
                              }
                          }
                       
                          static private void Merge(int[] A,int p,int q,int r) {
                              int n1=q-p+1;
                              int n2=r-q;
                              int[] L=new int[n1+1];
                              int[] R=new int[n2+1];
                              for(int i=0;i<n1;i++)
                                  L[i]=A[p+i];
                              L[n1]=10000;
                              for(int j=0;j<n2;j++)
                                  R[j]=A[q+1+j];
                              R[n2]=10000;
                              int i=0;
                              int j=0;
                              for(int k=p;k<=r;k++) {
                                  if(L[i]<R[j]) {
                                      A[k]=L[i];
                                      i++;
                                  } else {
                                      A[k]=R[j];
                                      j++;
                                  }
                              }
                          }
                       
                          public static void main(String[] args) {
                              int[] A={3,7,0,9,-1};
                              for(int i=0;i<A.length;i++)
                                  System.out.print(A[i]+" ");
                              System.out.println();
                              Sort(A);
                              System.out.println("Array ordinato: ");
                              for(int i=0;i<A.length;i++)
                                  System.out.print(A[i]+" ");
                          }
                      }

                        public class CountingSort {
                         
                            public static int[] Sort(int[] array, int range) {
                                int[] cont=new int[range+1];
                                for(int i=0;i<=range;i++)
                                    cont[i]=0;
                                for(int i=0;i<array.length;i++)
                                    cont[array[i]]++;
                                for(int i=1;i<=range;i++)
                                    cont[i]+=cont[i-1];
                                int[] sol=new int[array.length];
                                for(int i=array.length-1;i>=0;i--) {
                                    sol[cont[array[i]]-1]=array[i];
                                    cont[array[i]]--;
                                }
                                return sol;
                            }
                         
                            public static void main(String[] args) {
                                int[] A={1,6,3,0,1,4,5};

                                    System.out.print(A[i]+" ");
                                System.out.println();
                                int[] O=Sort(A,6);
                                for(int i=0;i<O.length;i++)
                                    System.out.print(O[i]+" ");
                                System.out.println();
                            }

                        package heapsort;
 
                        public class HeapSort {
                         
                            private static void Build(int[] array) {
                                for(int i=array.length/2;i>=0;i--)
                                    Heapify(array,i,array.length-1);
                            }
                         
                            private static void Heapify(int[] array,int i,int n) {
                                int max=i;
                                if((2*i+2)<n)
                                    if(array[2*i+1]<array[2*i+2])
                                        max=2*i+2;
                                    else
                                        max=2*i+1;
                                else
                                    if((2*i+1)<n)
                                        max=2*i+1;
                                if(array[i]<array[max]) {
                                    int temp=array[max];
                                    array[max]=array[i];
                                    array[i]=temp;
                                    Heapify(array,max,n);
                                }
                            }
                         
                            public static void Sort(int[] array) {
                                Build(array);
                                for(int i=array.length-1;i>=1;i--) {
                                    int temp=array[0];
                                    array[0]=array[i];
                                    array[i]=temp;
                                    Heapify(array,0,i-1);
                                }
                            }
                         
                            public static void main(String[] args) {
                                int[] array={3,12,-2,5,-1,9,1,8,0};
                                for(int i=0;i<array.length;i++)
                                    System.out.print(array[i]+" ");
                                System.out.println();
                                Sort(array);
                                System.out.println("Array ordinato: ");
                                for(int i=0;i<array.length;i++)
                                    System.out.print(array[i]+" ");
                            }
                        }
                      
                    

                        package quicksort;
 
                        public class QuickSort {
                         
                            static void Sort(int[] A) {
                                SortCore(A,0,A.length-1);
                            }
                         
                            private static void SortCore(int[] A,int p,int r) {
                                if(p < r) {
                                    int q=Partition(A,p,r);
                                    SortCore(A,p,q-1);
                                    SortCore(A,q+1,r);
                                }
                            }
                         
                            private static int Partition(int[] A,int p,int r) {
                                int pivot=A[r];
                                int i=p-1;
                                for(int j=p;j < r;j++) {
                                    if(A[j] <=pivot) {
                                        i++;
                                        int temp=A[i];
                                        A[i]=A[j];
                                        A[j]=temp;
                                    }
                                }
                                i++;
                                int temp=A[i];
                                A[i]=A[r];
                                A[r]=temp;
                                return i;
                            }
                         
                            public static void main(String[] args) {
                                int[] A={3,6,1,8,-3,7,-1};
                                for(int i=0;i < A.length;i++)
                                    System.out.print(A[i]+" ");
