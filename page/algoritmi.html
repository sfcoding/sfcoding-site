                      void insertion_sort (int *A, int n) {
                        int i, j, z;
                        for (i = 1; i < n; i++){
                          z = A[i], j = i-1;
                          while ( j >= 0 && A[j] > z ){
                            A[j+1] = A[j];
                            j--;
                          }
                          A[j+1] = z;
                        }
                        return;
                      }

                      void insertionSort(vector  & v){
                        uint size = v.size();
                        for(uint i = 1; i < size; ++i)
                        {
                          T z = v.at(i);
                          uint j;
                          for(j = i; j-- > 0 && v.at(j) > z; )
                            v.at(j+1) = v.at(j);
                          v.at(j+1) = z;
                        }
                      }
                       
                      template 
                      void merge(vector  & v, uint p, uint q, uint r){
                       
                        uint s1 = q - p + 2;
                        uint s2 = r - q + 1;
                        uint i = 0, j = 0;
                       
                        vector  v1(s1);
                        vector  v2(s2);
                       
                        for(; i < s1 - 1 ; ++i)
                          v1.at(i) = v.at(p + i);
                       
                        v1.at(i) = numeric_limits::max();
                       
                        for(i = 0; i < s2 - 1; ++i)
                          v2.at(i) = v.at(q + i + 1);
                       
                        v2.at(i) = numeric_limits::max();
                       
                        i = 0;
                       
                        for(uint k = p; k <= r; ++k)
                          v.at(k) = v1.at(i) < v2.at(j) ? v1.at(i++) : v2.at(j++);
                       
                      }
                       
                      template 
                      void mergeSort(vector  & v, uint p, uint r){
                       
                        if(p >= r) return;
                        int q = floor((p + r) / 2);
                        mergeSort(v, p, q);
                        mergeSort(v, q+1, r);
                        merge(v, p, q, r);
                       
                      }
                       
                      template 
                      void mergeSort(vector  & v){
                        mergeSort(v, 0, v.size()-1);
                      }
                       
                       
                      template 
                      int partition(vector  &v, uint p, uint r){
                        int i = p;
                        for(uint j = p; j < r; ++j)
                            if(v.at(j) <= v.at( r ))
                              swap(v.at(j), v.at(i++));
                        swap(v.at(i), v.at( r ) );
                        return i;
                      }
                       
                       
                      template 
                      void quickSort(vector  &v, uint p, uint r){
                        if(p >= r) return;
                        int q = partition(v,p,r);
                        quickSort(v, p, q-1);
                        quickSort(v, q+1, r);
                      }

                      void quickSort(vector  &v){
                        quickSort(v, 0, v.size()-1);

                          v[i] = 0;

                          v[I[i]]++;
                       
                        for(i = 1; i < limit; ++i)
                          v[i] += v[i-1];
                       
                        for(i = size; i-- > 0; )
                        {

                      void selectionSort(vector  & v){
                        for(int i = 0; i < v.size() - 1; i++)
                        {
                          int min = i;
                          for(int j = i+1; j < v.size(); j++)
                            if(v.at(j) < v.at(min))
                              min = j;
                          swap(v[min], v[i]);
                        }
                      }

                      class Heap{
                       
                        private:
                          vector  * v;
                          uint heapSize;
                       
                              public:
                       
                          static int parent(uint i){
                            return i/2;
                          }
                       
                          static int left(uint i){
                            return i*2 + 1;
                          }
                       
                          static int right(uint i){
                            return 2*(i+1);
                          }
                       
                          void maxHeapify(uint i){
                            uint max = i;
                            uint l = left(i);
                            uint r = right(i);
                            if(l < heapSize && v->at(max) < v->at(l))
                                max = l;
                            if(r < heapSize && v->at(max) < v->at( r ))
                                max = r;
                            if(max != i){
                              swap(v->at(i), v->at(max));
                              maxHeapify(max);
                            }
                          } 
                       
                          void buildMaxHeap(){
                            for(uint i = floor(v->size()/2)+1; i-- > 0; )
                                 maxHeapify(i);
                          }
                       
                          static void sort(vector  &v){
                       
                            Heap * hp = new Heap(v);
                       
                            for(uint i = v.size() - 1; i > 0; --i)
                            {
                              swap(v.at(0), v.at(i));
                              --hp -> heapSize;
                              hp -> maxHeapify(0);
                            }
                       
                            delete hp;
                          }
                       
                          T maximum(){
                            return v->at(0);
                          }
                       
                          T extractMaximum(){
                            int max = v->at(0);
                            v->at(0) = v->at(--heapSize);
                            maxHeapify(0);
                            return max;
                          }
                       
                          void increaseToKey(uint i, uint k)
                          {
                            if(k < v->at(i))
                            {
                              cout << "k dev'essere maggiore del vecchio valore!";
                              return;
                            }
                            v->at(i) = k;
                            uint par = parent(i)+1;
                            while(par-- > 0 && v->at(par) < v->at(i))
                            {
                              swap(v->at(i), v->at(par));
                              i = par;
                              par = parent(par)+1;
                            }
                          }
                       
                          void insert(T k){
                            v->push_back(numeric_limits::min());
                            increaseToKey(heapSize++, k);
                          }
                       
                          Heap(vector  &v){
                            this->v = &v;
                            this->heapSize = v.size();
                            buildMaxHeap();
                          }
                       
                      };
